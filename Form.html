function validateForm() { let isValid = true; document.querySelectorAll('.error').forEach(el => el.innerText = ''); document.querySelectorAll('.input-error').forEach(el => el.classList.remove('input-error')); const name = document.getElementById('fullname'); const email = document.getElementById('email'); const age = document.getElementById('age'); const city = document.getElementById('city'); const gender = document.querySelector('input[name="gender"]:checked'); const skills = document.querySelectorAll('input[name="skills"]:checked'); if (name.value.trim() === '') { document.getElementById('nameError').innerText = 'Name takna jaruri aahe.'; name.classList.add('input-error'); isValid = false; } else if (name.value.trim().length < 3) { document.getElementById('nameError').innerText = 'Name kamaan 3 aksharache pahije.'; name.classList.add('input-error'); isValid = false; } const emailPattern = /^[^\s@]+@[^\s@]+\.[^\s@]+$/; if (email.value.trim() === '') { document.getElementById('emailError').innerText = 'Email takna jaruri aahe.'; email.classList.add('input-error'); isValid = false; } else if (!emailPattern.test(email.value)) { document.getElementById('emailError').innerText = 'Email barobar taka.'; email.classList.add('input-error'); isValid = false; } // Age Validation if (age.value === '') { document.getElementById('ageError').innerText = 'Age takna jaruri aahe.'; age.classList.add('input-error'); isValid = false; } else if (age.value < 18) { document.getElementById('ageError').innerText = 'Age 18 peksha jast pahije.'; age.classList.add('input-error'); isValid = false; } if (city.value === '') { document.getElementById('cityError').innerText = 'City select kara.'; city.classList.add('input-error'); isValid = false; } if (!gender) { document.getElementById('genderError').innerText = 'Gender select kara.'; isValid = false; } if (skills.length === 0) { document.getElementById('skillsError').innerText = 'Kamaan 1 skill select kara.'; isValid = false; } return isValid; }
